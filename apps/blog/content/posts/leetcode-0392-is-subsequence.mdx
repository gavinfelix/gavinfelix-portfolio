### 代码实现（TypeScript）

```ts
function isSubsequence(s: string, t: string): string {
  let pointer = 0;

  for (let i = 0; i < t.length; i++) {
    if (pointer > s.length) return true;

    if (s[pointer] === t[i]) {
      pointer++;
    }
  }

  return pointer === s.length;
}
```
