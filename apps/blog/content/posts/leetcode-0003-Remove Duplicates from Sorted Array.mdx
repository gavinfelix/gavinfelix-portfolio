### 代码实现（TypeScript）

{/* Two-pointer in-place: slow = next unique write index, fast = scanner. */}

```ts
functionremoveDuplicates(nums:number[]):number {
if (nums.length ===0)return0;

let slow =1;

for (let fast =1; fast < nums.length; fast++) {
if (nums[fast] !== nums[fast -1]) {
      nums[slow] = nums[fast];
      slow++;
    }
  }

return slow;
}
```
