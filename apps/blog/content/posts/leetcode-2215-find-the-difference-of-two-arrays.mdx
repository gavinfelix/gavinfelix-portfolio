### 代码实现（TypeScript）

```ts
function findDifference(nums1: number[], nums2: number[]): number[][] {
  const set1 = new Set(nums1);
  const set2 = new Set(nums2);

  const arr1: number[] = [];
  const arr2: number[] = [];

  for (const n of set1) {
    if (!set2.has(n)) arr1.push(n);
  }

  for (const n of set2) {
    if (!set1.has(n)) arr2.push(n);
  }

  return [arr1, arr2];
}
```
