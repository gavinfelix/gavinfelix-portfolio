### 代码实现（TypeScript）

```ts
function intersection(nums1: number[], nums2: number[]): number[] {
  const set1 = new Set(nums1);
  const res = new Set<number>();

  for (let num of nums2) {
    if (set1.has(num)) res.add(num);
  }

  return Array.from(res);
}
```
